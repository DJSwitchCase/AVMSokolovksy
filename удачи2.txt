%include "io.inc"
section .data
Vec1 dd 7,2,3,8
section .text
global CMAIN
CMAIN:
    MOV eax, 4
    MOV CX, 3
    MOV edi, 0
  
    MOV ebx, 4
    MOV ax, [Vec1+edi]
    MOV bx, [Vec1+ebx]
    CMP ax, bx
    JS min1
    MOV cx, bx
    jmp aftermin1
    
    min1:
    MOV cx, ax
    MOV edx, ebx
    
    
    aftermin1:
    MOV ebx, 8
    MOV ax, [Vec1+ebx]
    CMP ax, cx
    
    
    JS min2
    jmp aftermin2
    
    min2:
    MOV cx, ax
    MOV edx, ebx
    
    
    aftermin2:
    
    MOV ebx, 12
    MOV ax, [Vec1+ebx]
    CMP ax, cx
    JS min3
    
    jmp letssort
    
    min3:
    MOV cx, ax
    MOV edx, ebx
    
    
    
    
    
    
    letssort:

    MOV ebx, [Vec1+edi]
    
    MOV [Vec1+edi], cx
    MOV [Vec1+edx],ebx
    
    
    
    
    
    

    MOV ax, [Vec1+0]
    PRINT_DEC 1, ax
    NEWLINE
    MOV ax, [Vec1+4]
    PRINT_DEC 1, ax
    NEWLINE
    MOV ax, [Vec1+8]
    PRINT_DEC 1, ax
    NEWLINE
    MOV ax, [Vec1+12]
    PRINT_DEC 1, ax
    NEWLINE
    ret
